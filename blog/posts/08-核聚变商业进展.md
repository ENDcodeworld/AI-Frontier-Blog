```python
# fusion/tokamak.py
import numpy as np
from typing import Dict

class TokamakPhysics:
    """托卡马克物理模型"""
    
    def __init__(self, major_radius=3.0, minor_radius=1.0, 
                 magnetic_field=12.0):
        """
        Args:
            major_radius: 大半径 (m)
            minor_radius: 小半径 (m)
            magnetic_field: 磁场强度 (T)
        """
        self.R = major_radius
        self.a = minor_radius
        self.B = magnetic_field
    
    def plasma_volume(self) -> float:
        """等离子体体积"""
        return 2 * np.pi**2 * self.R * self.a**2
    
    def plasma_pressure(self, temperature_keV=15, 
                       density=1e20) -> float:
        """
        等离子体压力
        
        Args:
            temperature_keV: 温度 (keV)
            density: 密度 (m^-3)
        """
        kB = 1.38e-23  # 玻尔兹曼常数
        eV_to_J = 1.6e-19
        
        T = temperature_keV * 1000 * eV_to_J / kB
        p = density * kB * T
        
        return p
    
    def beta(self, pressure=None) -> float:
        """
        比压 β (等离子体压力/磁压力)
        关键性能指标
        """
        if pressure is None:
            pressure = self.plasma_pressure()
        
        magnetic_pressure = self.B**2 / (2 * 4*np.pi*1e-7)
        
        return pressure / magnetic_pressure
    
    def triple_product(self, temperature_keV=15,
                      density=1e20, 
                      confinement_time=3.0) -> float:
        """
        三重积 nTτ
        点火条件：> 5e21 keV·s/m³
        """
        return density * temperature_keV * confinement_time
    
    def fusion_power_density(self) -> float:
        """聚变功率密度"""
        # D-T 反应截面简化模型
        T = 15  # keV
        n = 1e20  # m^-3
        
        # 反应率系数 (简化)
        <sigma_v> = 1e-22 * T**2
        
        # 每次反应能量 17.6 MeV
        E_fusion = 17.6 * 1.6e-13  # J
        
        # 功率密度
        P = n**2 * <sigma_v> * E_fusion / 4
        
        return P
    
    def q_engineering(self, fusion_power_mw=500, 
                     heating_power_mw=100) -> float:
        """
        工程增益 Q
        Q = 聚变功率 / 加热功率
        """
        return fusion_power_mw / heating_power_mw
    
    def q_scientific(self, fusion_power_mw=500,
                    total_input_mw=150) -> float:
        """
        科学增益 Q
        考虑所有输入功率
        """
        return fusion_power_mw / total_input_mw
```

```python
class InertialConfinementPhysics:
    """惯性约束聚变物理"""
    
    def __init__(self, laser_energy_mj=2000):
        self.laser_energy = laser_energy_mj * 1e6  # J
    
    def implosion_velocity(self, fuel_mass_kg=1e-7) -> float:
        """
        内爆速度
        需要 > 3e5 m/s 才能点火
        """
        # 动能 = 0.5 * m * v²
        # 假设 10% 激光能量转换为动能
        kinetic_energy = 0.1 * self.laser_energy
        
        velocity = np.sqrt(2 * kinetic_energy / fuel_mass_kg)
        return velocity
    
    def areal_density(self, fuel_radius_m=1e-4,
                     fuel_density_kg_m3=1000) -> float:
        """
        面密度 ρR
        点火条件：> 0.3 g/cm²
        """
        rho_R = fuel_density_kg_m3 * fuel_radius_m
        return rho_R * 100  # g/cm²
    
    def gain(self, fusion_yield_mj=3000) -> float:
        """
        能量增益
        NIF 2022: 2.05 MJ 输入 → 3.15 MJ 输出 (Q=1.5)
        """
        return fusion_yield_mj / (self.laser_energy / 1e6)

        """
        热点条件
        点火所需：
        - 温度：> 5 keV
        - 密度：> 100 g/cm³
        - ρR: > 0.3 g/cm²
```

```python
# fusion/ai_control.py
import torch
import torch.nn as nn
import numpy as np

class PlasmaController(nn.Module):
    """
    基于深度强化学习的等离子体控制器
    控制磁场、加热、燃料注入等
    """
    
    def __init__(self, n_sensors=100, n_actuators=50):
        super().__init__()
        
        # 状态编码器
        self.encoder = nn.Sequential(
            nn.Linear(n_sensors, 512),
            nn.ReLU(),
            nn.Linear(512, 256),
            nn.ReLU()
        )
        
        # Actor (策略)
        self.actor = nn.Sequential(
            nn.Linear(256, 128),
            nn.ReLU(),
            nn.Linear(128, n_actuators),
            nn.Tanh()  # 输出范围 [-1, 1]
        )
        
        # Critic (价值)
        self.critic = nn.Sequential(
            nn.Linear(256, 128),
            nn.ReLU(),
            nn.Linear(128, 1)
        )
    
    def forward(self, state):
        """
        Args:
            state: 传感器读数 (batch, n_sensors)
            
        Returns:
            action: 控制指令 (batch, n_actuators)
            value: 状态价值估计
        """
        encoded = self.encoder(state)
        action = self.actor(encoded)
        value = self.critic(encoded)
        
        return action, value
    
    def select_action(self, state, deterministic=False):
        """选择动作"""
        self.eval()
        with torch.no_grad():
            action, _ = self.forward(state.unsqueeze(0))
            
            if deterministic:
                return action.squeeze().numpy()
            else:
                # 添加探索噪声
                noise = torch.randn_like(action) * 0.1
                return (action + noise).squeeze().numpy()


class PlasmaDisruptionPredictor(nn.Module):
    """
    等离子体破裂预测
    提前预警，避免设备损坏
    """
    
    def __init__(self, sequence_length=100, n_features=50):
        super().__init__()
        
        self.lstm = nn.LSTM(
            input_size=n_features,
            hidden_size=256,
            num_layers=3,
            batch_first=True,
            dropout=0.3
        )
        
        self.attention = nn.MultiheadAttention(
            embed_dim=256,
            num_heads=8
        )
        
        self.classifier = nn.Sequential(
            nn.Linear(256, 64),
            nn.ReLU(),
            nn.Dropout(0.3),
            nn.Linear(64, 2)  # 正常/破裂
        )
    
    def forward(self, sequence):
        """
        Args:
            sequence: 时间序列数据 (batch, seq_len, n_features)
            
        Returns:
            disruption_probability: 破裂概率
            time_to_disruption: 预计破裂时间 (ms)
        """
        # LSTM 编码
        lstm_out, (h_n, c_n) = self.lstm(sequence)
        
        # 注意力机制
        attn_out, _ = self.attention(
            lstm_out.transpose(0, 1),
            lstm_out.transpose(0, 1),
            lstm_out.transpose(0, 1)
        )
        
        # 使用注意力加权表示
        context = attn_out.mean(dim=0)
        
        # 分类
        logits = self.classifier(context)
        probs = torch.softmax(logits, dim=-1)
        
        disruption_prob = probs[:, 1]
        
        # 估计破裂时间 (回归头)
        time_to_disruption = self._time_head(context)
        
        return disruption_prob, time_to_disruption
```

```python
# fusion/magnet_optimization.py
import torch
import torch.nn as nn
from torchdiffeq import odeint

class MagnetFieldSimulator:
    """
    磁场仿真器
    使用神经网络加速 Biot-Savart 定律计算
    """
    
    def __init__(self, n_coils=18):
        self.n_coils = n_coils
        
        # 神经网络替代传统数值计算
        self.field_network = nn.Sequential(
            nn.Linear(6, 128),  # 输入：位置 (3) + 线圈参数 (3)
            nn.ReLU(),
            nn.Linear(128, 128),
            nn.ReLU(),
            nn.Linear(128, 3)  # 输出：磁场向量
        )
    
    def calculate_field(self, positions, coil_params):
        """
        计算磁场
        
        Args:
            positions: 空间位置 (N, 3)
            coil_params: 线圈参数 (n_coils, 3)
        """
        # 传统方法：Biot-Savart 定律
        # B = μ₀/4π * I * dl × r / r³
        
        # 神经网络加速版本
        field = self.field_network(
            torch.cat([positions, coil_params.view(-1, 3).repeat(len(positions), 1)], dim=1)
        )
        
        return field
    
    def optimize_coil_configuration(self, target_field):
        """
        优化线圈配置以产生目标磁场
        使用梯度下降
        """
        coil_params = torch.randn(self.n_coils, 3, requires_grad=True)
        optimizer = torch.optim.Adam([coil_params], lr=0.01)
        
        for step in range(1000):
            field = self.calculate_field(self.probe_positions, coil_params)
            loss = torch.mean((field - target_field)**2)
            
            optimizer.zero_grad()
            loss.backward()
            optimizer.step()
        
        return coil_params.detach()
```

```python
# fusion/materials.py
from matminer.featurizers.composition import CompositionFeaturizer
from sklearn.ensemble import RandomForestRegressor

class FusionMaterialDiscovery:
    """
    聚变堆材料发现
    第一壁材料、偏滤器材料等
    """
    
    def __init__(self):
        self.featurizer = CompositionFeaturizer()
        self.model = RandomForestRegressor(n_estimators=100)
        
        # 关键性能指标
        self.properties = [
            'melting_point',      # 熔点
            'thermal_conductivity', # 热导率
            'neutron_damage_tolerance', # 中子损伤容限
            'tritium_retention',  # 氚滞留
            'activation'          # 活化
        ]
    
    def train(self, materials_data):
        """训练材料预测模型"""
        X = self.featurizer.featurize(materials_data['composition'])
        
        for prop in self.properties:
            y = materials_data[prop]
            self.model.fit(X, y)
    
    def screen_candidates(self, candidate_compositions):
        """
        筛选候选材料
        
        Returns:
            按综合性能排序的材料列表
        """
        X = self.featurizer.featurize(candidate_compositions)
        
        predictions = {}
        for prop in self.properties:
            predictions[prop] = self.model.predict(X)
        
        # 综合评分
        scores = self._calculate_composite_score(predictions)
        
        # 排序
        ranked = sorted(
            zip(candidate_compositions, scores),
            key=lambda x: -x[1]
        )
        
        return ranked
    
    def _calculate_composite_score(self, predictions):
        """计算综合性能评分"""
        # 归一化各性能指标
        normalized = {}
        for prop, values in predictions.items():
            min_val, max_val = values.min(), values.max()
            normalized[prop] = (values - min_val) / (max_val - min_val + 1e-6)
        
        # 加权求和 (权重根据重要性)
        weights = {
            'melting_point': 0.3,
            'thermal_conductivity': 0.2,
            'neutron_damage_tolerance': 0.25,
            'tritium_retention': 0.15,
            'activation': 0.1
        }
        
        scores = sum(
            weights[prop] * normalized[prop] 
            for prop in self.properties
        )
        
        return scores
```

核聚变正在从"永远还需要 50 年"变成"未来 10 年可见"。AI 技术的加入加速了这一进程，让等离子体控制、材料发现、系统设计变得更加高效。

无论最终哪种技术路线胜出，人类向"无限清洁能源"的迈进已经不可逆转。这不仅是技术革命，更是文明级别的跃迁。
