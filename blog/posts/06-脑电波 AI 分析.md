```python
# eeg/data_loader.py
import mne
import numpy as np
from typing import List, Tuple

class EEGDataLoader:
    """EEG 数据加载器"""
    
    def __init__(self, sampling_rate=256, n_channels=32):
        self.sampling_rate = sampling_rate
        self.n_channels = n_channels
        
        # 标准 10-20 系统电极位置
        self.channel_names = [
            'Fp1', 'Fp2', 'F7', 'F3', 'Fz', 'F4', 'F8',
            'T3', 'C3', 'Cz', 'C4', 'T4',
            'T5', 'P3', 'Pz', 'P4', 'T6',
            'O1', 'O2',
            # 参考电极
            'A1', 'A2'
        ]
    
    def load_edf(self, file_path: str) -> mne.io.Raw:
        """加载 EDF 格式 EEG 数据"""
        raw = mne.io.read_raw_edf(file_path, preload=True)
        raw.filter(0.5, 45)  # 带通滤波
        return raw
    
    def load_csv(self, file_path: str) -> np.ndarray:
        """加载 CSV 格式数据"""
        data = np.loadtxt(file_path, delimiter=',')
        return data
    
    def extract_epochs(self, raw: mne.io.Raw, events: np.ndarray, 
                      tmin=-0.2, tmax=0.8) -> mne.Epochs:
        """
        提取事件相关电位 (ERP)
        
        Args:
            raw: 原始数据
            events: 事件标记
            tmin:  epoch 开始时间 (相对事件)
            tmax:  epoch 结束时间
        """
        epochs = mne.Epochs(
            raw, events,
            tmin=tmin, tmax=tmax,
            baseline=(None, 0),  # 基线校正
            preload=True
        )
        return epochs
    
    def extract_features(self, data: np.ndarray) -> dict:
        """
        提取传统特征
        
        Returns:
            {
                'psd': 功率谱密度,
                'band_power': 频段功率,
                'asymmetry': 半球不对称性,
                ...
            }
        """
        features = {}
        
        # 功率谱密度
        freqs, psd = self._compute_psd(data)
        features['psd'] = psd
        
        # 频段功率
        band_powers = {}
        for band in EEGBand:
            mask = (freqs >= band.low) & (freqs < band.high)
            band_powers[band.name] = np.mean(psd[:, mask], axis=1)
        features['band_power'] = band_powers
        
        # 半球不对称性
        features['asymmetry'] = self._compute_asymmetry(psd)
        
        return features
```

```python
# models/eegnet.py
import torch
import torch.nn as nn
import torch.nn.functional as F

class EEGNet(nn.Module):
    """
    EEGNet: 紧凑卷积神经网络用于脑电分类
    参考：Lawhern et al., 2018
    """
    
    def __init__(self, n_channels=32, n_samples=256, n_classes=4, 
                 dropout_rate=0.5):
        super().__init__()
        
        # Block 1: 时间卷积 + 空间卷积
        self.conv1 = nn.Conv2d(1, 8, (1, 64), padding=(0, 32), bias=False)
        self.bn1 = nn.BatchNorm2d(8)
        
        self.depthwise_conv = nn.Conv2d(8, 16, (n_channels, 1), 
                                        groups=8, bias=False)
        self.bn2 = nn.BatchNorm2d(16)
        
        self.pool1 = nn.AvgPool2d((1, 4))
        self.dropout1 = nn.Dropout(dropout_rate)
        
        # Block 2: 可分离卷积
        self.separable_conv = nn.Conv2d(16, 16, (1, 16), 
                                        padding=(0, 8), bias=False)
        self.bn3 = nn.BatchNorm2d(16)
        
        self.pool2 = nn.AvgPool2d((1, 4))
        self.dropout2 = nn.Dropout(dropout_rate)
        
        # 分类层
        self.flatten = nn.Flatten()
        self.fc = nn.Linear(16 * 16, n_classes)
    
    def forward(self, x):
        # x shape: (batch, 1, n_channels, n_samples)
        
        # Block 1
        x = self.conv1(x)
        x = self.bn1(x)
        x = F.elu(x)
        
        x = self.depthwise_conv(x)
        x = self.bn2(x)
        x = F.elu(x)
        
        x = self.pool1(x)
        x = self.dropout1(x)
        
        # Block 2
        x = self.separable_conv(x)
        x = self.bn3(x)
        x = F.elu(x)
        
        x = self.pool2(x)
        x = self.dropout2(x)
        
        # 分类
        x = self.flatten(x)
        x = self.fc(x)
        
        return x


class EEGNetV2(nn.Module):
    """改进版 EEGNet"""
    
    def __init__(self, n_channels=32, n_samples=256, n_classes=4,
                 F1=8, D=2, F2=16, dropout_rate=0.25):
        super().__init__()
        
        self.F1 = F1
        self.D = D
        self.F2 = F2
        
        # Block 1
        self.temporal_conv = nn.Conv2d(1, F1, (1, 64), padding=(0, 32), bias=False)
        self.bn1 = nn.BatchNorm2d(F1)
        
        self.depthwise_conv = nn.Conv2d(F1, F1*D, (n_channels, 1), 
                                        groups=F1, bias=False)
        self.bn2 = nn.BatchNorm2d(F1*D)
        
        self.pool1 = nn.AvgPool2d((1, 4))
        self.dropout1 = nn.Dropout(dropout_rate)
        
        # Block 2
        self.separable_conv = nn.Conv2d(F1*D, F2, (1, 16), 
                                        padding=(0, 8), bias=False)
        self.bn3 = nn.BatchNorm2d(F2)
        
        self.pool2 = nn.AvgPool2d((1, 4))
        self.dropout2 = nn.Dropout(dropout_rate)
        
        # 分类
        self.fc = nn.Linear(F2 * 16, n_classes)
    
    def forward(self, x):
        x = self.temporal_conv(x)
        x = self.bn1(x)
        x = F.elu(x)
        
        x = self.depthwise_conv(x)
        x = self.bn2(x)
        x = F.elu(x)
        
        x = self.pool1(x)
        x = self.dropout1(x)
        
        x = self.separable_conv(x)
        x = self.bn3(x)
        x = F.elu(x)
        
        x = self.pool2(x)
        x = self.dropout2(x)
        
        x = x.view(x.size(0), -1)
        x = self.fc(x)
        
        return x
```

### 3.2 Transformer 模型

```python
# models/eeg_transformer.py
import torch
import torch.nn as nn
import math

class PositionalEncoding(nn.Module):
    """位置编码"""
    
    def __init__(self, d_model, max_len=5000):
        super().__init__()
        
        pe = torch.zeros(max_len, d_model)
        position = torch.arange(0, max_len, dtype=torch.float).unsqueeze(1)
        div_term = torch.exp(torch.arange(0, d_model, 2).float() * 
                            (-math.log(10000.0) / d_model))
        
        pe[:, 0::2] = torch.sin(position * div_term)
        pe[:, 1::2] = torch.cos(position * div_term)
        
        self.register_buffer('pe', pe)
    
    def forward(self, x):
        return x + self.pe[:x.size(1), :].unsqueeze(0)


class EEGTransformer(nn.Module):
    """基于 Transformer 的 EEG 分类模型"""
    
    def __init__(self, n_channels=32, n_samples=256, n_classes=4,
                 d_model=128, nhead=8, num_layers=4, dropout=0.1):
        super().__init__()
        
        # 输入投影
        self.input_projection = nn.Linear(n_samples, d_model)
        
        # 位置编码
        self.pos_encoder = PositionalEncoding(d_model)
        
        # Transformer Encoder
        encoder_layer = nn.TransformerEncoderLayer(
            d_model=d_model,
            nhead=nhead,
            dim_feedforward=d_model*4,
            dropout=dropout,
            batch_first=True
        )
        
        self.transformer = nn.TransformerEncoder(
            encoder_layer,
            num_layers=num_layers
        )
        
        # 分类头
        self.classifier = nn.Sequential(
            nn.Linear(d_model, 64),
            nn.ReLU(),
            nn.Dropout(dropout),
            nn.Linear(64, n_classes)
        )
    
    def forward(self, x):
        # x shape: (batch, n_channels, n_samples)
        
        # 投影到 d_model
        x = self.input_projection(x)  # (batch, n_channels, d_model)
        
        # 添加位置编码
        x = self.pos_encoder(x)
        
        # Transformer
        x = self.transformer(x)
        
        # 全局平均池化
        x = x.mean(dim=1)
        
        # 分类
        x = self.classifier(x)
        
        return x
```

### 3.3 图神经网络

```python
# models/eeg_gnn.py
import torch
import torch.nn as nn
import torch.nn.functional as F
from torch_geometric.nn import GCNConv, GATConv

class EEGGraphNN(nn.Module):
    """
    基于图神经网络的 EEG 分类
    将电极视为图节点，利用大脑连接性
    """
    
    def __init__(self, n_channels=32, n_features=64, n_classes=4):
        super().__init__()
        
        # 电极间的空间连接 (基于 10-20 系统距离)
        self.edge_index = self._create_electrode_graph()
        
        # GCN 层
        self.gcn1 = GCNConv(n_features, 64)
        self.gcn2 = GCNConv(64, 32)
        
        # 分类层
        self.fc = nn.Linear(32 * n_channels, n_classes)
    
    def _create_electrode_graph(self):
        """创建电极连接图"""
        # 基于电极物理位置创建邻接
        # 这里简化处理
        edge_index = torch.tensor([
            [0, 1, 2, 3],
            [1, 2, 3, 0]
        ], dtype=torch.long)
        return edge_index
    
    def forward(self, x, edge_index=None):
        # x shape: (batch, n_channels, n_features)
        
        if edge_index is None:
            edge_index = self.edge_index
        
        batch_size = x.size(0)
        
        # 对每个样本应用 GCN
        outputs = []
        for i in range(batch_size):
            xi = x[i]  # (n_channels, n_features)
            
            # GCN 层
            h = F.relu(self.gcn1(xi, edge_index))
            h = F.dropout(h, p=0.2, training=self.training)
            h = F.relu(self.gcn2(h, edge_index))
            
            outputs.append(h.flatten())
        
        x = torch.stack(outputs)
        x = self.fc(x)
        
        return x
```

```python
# applications/motor_imagery.py
import torch
from models.eegnet import EEGNet

class MotorImageryClassifier:
    """运动想象分类器"""
    
    def __init__(self, n_channels=32, n_classes=4):
        self.model = EEGNet(n_channels=n_channels, n_classes=n_classes)
        self.class_names = ['left_hand', 'right_hand', 'foot', 'tongue']
    
    def train(self, train_loader, val_loader, epochs=100):
        """训练模型"""
        device = torch.device('cuda' if torch.cuda.is_available() else 'cpu')
        self.model.to(device)
        
        criterion = nn.CrossEntropyLoss()
        optimizer = torch.optim.Adam(self.model.parameters(), lr=0.001)
        scheduler = torch.optim.lr_scheduler.ReduceLROnPlateau(
            optimizer, mode='min', factor=0.5, patience=10
        )
        
        best_acc = 0
        
        for epoch in range(epochs):
            # 训练
            self.model.train()
            train_loss = 0
            correct = 0
            total = 0
            
            for data, target in train_loader:
                data, target = data.to(device), target.to(device)
                
                optimizer.zero_grad()
                output = self.model(data)
                loss = criterion(output, target)
                loss.backward()
                optimizer.step()
                
                train_loss += loss.item()
                pred = output.argmax(dim=1)
                correct += pred.eq(target).sum().item()
                total += target.size(0)
            
            train_acc = correct / total
            
            # 验证
            self.model.eval()
            val_acc = self._evaluate(val_loader, device)
            
            scheduler.step(1 - val_acc)
            
            print(f'Epoch {epoch+1}: Train Acc={train_acc:.4f}, Val Acc={val_acc:.4f}')
            
            if val_acc > best_acc:
                best_acc = val_acc
                torch.save(self.model.state_dict(), 'best_mi_model.pth')
        
        return best_acc
    
    def predict(self, eeg_data: np.ndarray) -> tuple:
        """
        预测运动想象类别
        
        Args:
            eeg_data: EEG 数据 (n_channels, n_samples)
            
        Returns:
            (class_name, confidence)
        """
        device = torch.device('cuda' if torch.cuda.is_available() else 'cpu')
        self.model.to(device)
        self.model.eval()
        
        with torch.no_grad():
            data = torch.FloatTensor(eeg_data).unsqueeze(0).unsqueeze(0).to(device)
            output = self.model(data)
            prob = F.softmax(output, dim=1)
            pred = output.argmax(dim=1)
            
            return self.class_names[pred.item()], prob[0][pred].item()
```

```python
        提取情绪相关特征

            {
                'frontal_alpha_asymmetry': ...,
                'beta_gamma_ratio': ...,
                ...
            }
        """
        features = {}

        f3_power = self._band_power(eeg_data, 'F3', 'alpha')
        f4_power = self._band_power(eeg_data, 'F4', 'alpha')

        beta_power = self._global_band_power(eeg_data, 'beta')
        gamma_power = self._global_band_power(eeg_data, 'gamma')
        features['beta_gamma_ratio'] = beta_power / (gamma_power + 1e-6)
        
        return features
    
    def predict_emotion(self, eeg_data: np.ndarray) -> dict:
        """预测情绪状态"""
        self.model.eval()
        
        with torch.no_grad():
            data = torch.FloatTensor(eeg_data).unsqueeze(0)
            output = self.model(data)
            prob = F.softmax(output, dim=1)
            pred = output.argmax(dim=1)
            
            return {
                'emotion': self.emotions[pred.item()],
                'confidence': prob[0][pred].item(),
                'probabilities': {
                    self.emotions[i]: prob[0][i].item() 
                    for i in range(4)
                }
```

```python
import torch
from torch.utils.data import Dataset, DataLoader

class SeizureDetectionDataset(Dataset):
    """癫痫检测数据集"""
    
    def __init__(self, eeg_segments, labels, seq_length=256*10):
        self.eeg_segments = eeg_segments
        self.labels = labels
        self.seq_length = seq_length
    
    def __len__(self):
        return len(self.eeg_segments)
    
    def __getitem__(self, idx):
        segment = self.eeg_segments[idx]
        label = self.labels[idx]
        
        # 截断或填充到固定长度
        if len(segment) > self.seq_length:
            segment = segment[:self.seq_length]
        else:
            pad_length = self.seq_length - len(segment)
            segment = np.pad(segment, ((0, pad_length), (0, 0)))
        
        return torch.FloatTensor(segment), torch.LongTensor([label])[0]


class SeizureDetector:
    """癫痫发作检测器"""
    
    def __init__(self):
        self.model = EEGNet(n_channels=32, n_classes=2)  # 发作/正常
    
    def detect_seizure(self, eeg_segment: np.ndarray, 
                      threshold=0.7) -> dict:
        """
        检测癫痫发作
        
        Args:
            eeg_segment: EEG 片段 (n_channels, n_samples)
            threshold: 判定阈值
            
        Returns:
            {
                'is_seizure': bool,
                'probability': float,
                'confidence': str
            }
        """
        self.model.eval()
        
        with torch.no_grad():
            data = torch.FloatTensor(eeg_segment).unsqueeze(0).unsqueeze(0)
            output = self.model(data)
            prob = F.softmax(output, dim=1)[0][1].item()  # 发作类概率
            
            return {
                'is_seizure': prob > threshold,
                'probability': prob,
                'confidence': 'high' if prob > 0.9 else 'medium' if prob > 0.7 else 'low'
            }
    
    def continuous_monitoring(self, eeg_stream, window_size=10,
                             step_size=1) -> list:
        """
        连续监测 EEG 流
        
        Args:
            eeg_stream: EEG 数据流
            window_size: 窗口大小 (秒)
            step_size: 步长 (秒)
            
        Returns:
            检测结果列表
        """
        results = []
        sample_rate = 256
        window_samples = window_size * sample_rate
        step_samples = step_size * sample_rate
        
        for i in range(0, len(eeg_stream) - window_samples, step_samples):
            segment = eeg_stream[i:i+window_samples]
            result = self.detect_seizure(segment)
            result['timestamp'] = i / sample_rate
            results.append(result)
        
        return results
```

```python
        Returns:

        beta_power = self._band_power(eeg_data, channels, 'beta')

        # 转换为 0-100 分数
        # 比率越低，注意力越高
        attention_score = 100 / (1 + theta_beta_ratio)
        
        return min(100, max(0, attention_score))
    
    def classify_attention_level(self, attention_index: float) -> str:
        """分类注意力水平"""
        if attention_index >= 70:
            return 'HIGH'
        elif attention_index >= 40:
            return 'MEDIUM'
        else:
            return 'LOW'
```

```python
# realtime/stream_processor.py
import asyncio
from collections import deque
import numpy as np

class EEGStreamProcessor:
    """EEG 实时流处理器"""
    
    def __init__(self, sampling_rate=256, buffer_size=10):
        self.sampling_rate = sampling_rate
        self.buffer_size = buffer_size  # 秒
        
        # 环形缓冲区
        self.buffer = deque(
            maxlen=sampling_rate * buffer_size
        )
        
        # 锁
        self.lock = asyncio.Lock()
    
    async def add_sample(self, sample: np.ndarray):
        """添加新样本"""
        async with self.lock:
            self.buffer.append(sample)
    
    async def get_window(self, window_size: float) -> np.ndarray:
        """获取指定时间窗口的数据"""
        async with self.lock:
            n_samples = int(window_size * self.sampling_rate)
            
            if len(self.buffer) < n_samples:
                return None
            
            data = np.array(list(self.buffer)[-n_samples:])
            return data
    
    async def process_stream(self, model, callback, window_size=2.0):
        """
        实时处理流
        
        Args:
            model: 预测模型
            callback: 结果回调函数
            window_size: 处理窗口 (秒)
        """
        while True:
            data = await self.get_window(window_size)
            
            if data is not None:
                # 预处理
                data = self._preprocess(data)
                
                # 预测
                result = await self._predict(model, data)
                
                # 回调
                await callback(result)
            
            await asyncio.sleep(0.1)  # 100ms 间隔
```

```python
# realtime/visualization.py
import matplotlib.pyplot as plt
from matplotlib.animation import FuncAnimation
import numpy as np

class EEGVisualizer:
    """EEG 实时可视化"""
    
    def __init__(self, n_channels=32):
        self.n_channels = n_channels
        self.fig, self.axes = plt.subplots(n_channels, 1, figsize=(12, 16))
        self.lines = []
        
        for i, ax in enumerate(self.axes):
            line, = ax.plot([], [])
            self.lines.append(line)
            ax.set_ylim(-100, 100)
            ax.set_yticks([])
            ax.set_title(f'Channel {i}')
        
        plt.tight_layout()
    
    def update(self, data):
        """更新显示"""
        for i, line in enumerate(self.lines):
            if i < len(data):
                line.set_data(range(len(data[i])), data[i])
                ax = self.axes[i]
                ax.set_xlim(0, len(data[i]))
        
        return self.lines
    
    def start(self, stream_processor):
        """启动实时显示"""
        def animate(frame):
            data = asyncio.run(stream_processor.get_window(2.0))
            if data is not None:
                return self.update(data)
            return self.lines
        
        ani = FuncAnimation(self.fig, animate, interval=100)
        plt.show()
```
