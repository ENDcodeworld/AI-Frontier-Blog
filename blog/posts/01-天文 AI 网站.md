## 三、核心功能实现

### 3.1 AI 星座识别

这是项目的核心功能。用户上传星空照片，AI 自动识别其中的星座、行星、深空天体等。

**实现思路：**

1. 使用预训练的视觉模型提取图像特征
2. 结合天文星表进行匹配
3. 输出识别结果和详细信息

**代码示例：**

```python
# astro_ai/recognizer.py
import torch
from transformers import AutoModelForImageClassification
from astropy.coordinates import SkyCoord
import numpy as np

class ConstellationRecognizer:
    def __init__(self, model_name="qwen-vl"):
        self.model = AutoModelForImageClassification.from_pretrained(model_name)
        self.star_catalog = self._load_star_catalog()
    
    def recognize(self, image_path, location=None, timestamp=None):
        """
        识别星空照片中的星座和天体
        
        Args:
            image_path: 照片路径
            location: 拍摄地点 (经纬度)
            timestamp: 拍摄时间
            
        Returns:
            dict: 识别结果
        """
        # 1. 图像预处理
        image = self._preprocess_image(image_path)
        
        # 2. AI 模型推理
        features = self.model(image)
        
        # 3. 星图匹配
        matches = self._match_star_catalog(features, location, timestamp)
        
        # 4. 星座识别
        constellations = self._identify_constellations(matches)
        
        return {
            "constellations": constellations,
            "stars": matches["stars"],
            "planets": matches["planets"],
            "deep_sky_objects": matches["dso"]
        }
    
    def _match_star_catalog(self, features, location, timestamp):
        """与天文星表进行匹配"""
        # 根据拍摄时间和地点计算可见天区
        if location and timestamp:
            visible_region = self._calc_visible_region(location, timestamp)
        else:
            visible_region = self.star_catalog
        
        # 特征匹配
        matches = []
        for star in visible_region:
            similarity = self._calculate_similarity(features, star)
            if similarity > 0.85:
                matches.append({
                    "name": star["name"],
                    "magnitude": star["magnitude"],
                    "coordinates": star["coordinates"],
                    "confidence": similarity
                })
        
        return {"stars": sorted(matches, key=lambda x: -x["confidence"])[:50]}
```

### 3.2 实时星图生成

基于用户的位置和时间，生成个性化的实时星图。

```python
# astro_ai/star_map.py
from astropy.time import Time
from astropy.coordinates import EarthLocation, AltAz, SkyCoord
import astropy.units as u

class RealTimeStarMap:
    def __init__(self):
        self.star_catalog = self._load_hipparcos_catalog()
    
    def generate(self, latitude, longitude, timestamp, fov=60):
        """
        生成实时星图
        
        Args:
            latitude: 纬度
            longitude: 经度
            timestamp: 观测时间
            fov: 视场角 (度)
            
        Returns:
            dict: 星图数据
        """
        # 创建观测者位置
        location = EarthLocation(lat=latitude*u.deg, 
                                  lon=longitude*u.deg)
        time = Time(timestamp)
        
        # 计算天体的地平坐标
        visible_objects = []
        for star in self.star_catalog:
            if star["magnitude"] > 6:  # 只保留肉眼可见的星
                continue
                
            coord = SkyCoord(ra=star["ra"], dec=star["dec"])
            altaz = coord.transform_to(AltAz(obstime=time, location=location))
            
            if altaz.alt > 0 * u.deg:  # 地平线以上
                visible_objects.append({
                    "name": star["name"],
                    "magnitude": star["magnitude"],
                    "altitude": altaz.alt.value,
                    "azimuth": altaz.az.value,
                    "ra": star["ra"],
                    "dec": star["dec"]
                })
        
        return {
            "timestamp": timestamp,
            "location": {"lat": latitude, "lon": longitude},
            "objects": visible_objects,
            "constellations": self._identify_constellations(visible_objects)
        }
```

### 3.3 天文知识问答

集成大语言模型，提供智能天文知识问答。

```python
# astro_ai/qa_bot.py
from openai import OpenAI

class AstroQABot:
    def __init__(self, api_key, model="qwen-plus"):
        self.client = OpenAI(api_key=api_key)
        self.model = model
        self.system_prompt = """
        你是一位专业的天文科普助手，擅长用通俗易懂的语言解释天文现象。
        回答要准确、有趣，适当使用比喻帮助理解。
        如果不确定，请诚实说明。
        """
    
    def ask(self, question, context=None):
        """回答天文问题"""
        messages = [
            {"role": "system", "content": self.system_prompt},
        ]
        
        if context:
            messages.append({"role": "user", "content": f"背景信息：{context}"})
        
        messages.append({"role": "user", "content": question})
        
        response = self.client.chat.completions.create(
            model=self.model,
            messages=messages,
            temperature=0.7,
            max_tokens=1000
        )
        
        return response.choices[0].message.content
```

    const renderer = new THREE.WebGLRenderer({ antialias: true });
    
    // 创建星星
    starData.objects.forEach(star => {
      const geometry = new THREE.SphereGeometry(0.1, 8, 8);
      const material = new THREE.MeshBasicMaterial({
        color: getStarColor(star.magnitude),
        transparent: true,
        opacity: 1 - star.magnitude / 10
      });
      const sphere = new THREE.Mesh(geometry, material);
      
      // 将球坐标转换为笛卡尔坐标
